Debugging in Microsoft Visual Studio 2008

Add these lines to the [Visualizer] section in VSROOT\Common7\packages\debugger\autoexp.dat for Visual Studio for proper debugging in C++ 

; Xsens specific
XsDeviceId {
    preview     ( [$c.m_deviceId,x] )
    children    ( #( [raw members]: [$c,!] ) )
}

XsArrayImpl<*,*> {
    preview     ( #( [size]: [$e.m_size], [stze]: ": ", #array( expr: (($T1 *)$e.m_data)[$i], size: $e.m_size ) ) )
    children    ( #( [raw members]: [$c,!], [size]: [$c.m_size], #array ( expr: (($T1 *)$c.m_data)[$i], size: $c.m_size ) ) )
}
